# 실습 데이터
```SQL
DROP TABLE ROWNUM_T1 PURGE;


-- ROWNUM_T1 테이블 생성 

CREATE TABLE ROWNUM_T1 AS
SELECT LEVEL AS C1, CHR(65+MOD(LEVEL,26)) AS C2
  FROM DUAL CONNECT BY LEVEL <= 500000 ;

-- 각 칼럼에 인덱스 생성 및 통계정보 수집
CREATE INDEX IDX_01 ON ROWNUM_T1 ( C2, C1 ) ;









DROP TABLE ROWNUM_T2 PURGE ;


-- ROWNUM_T2 테이블 생성

CREATE TABLE ROWNUM_T2
AS
SELECT LEVEL AS C1, CHR(65+MOD(LEVEL,26)) AS C2, LEVEL+99999 AS C3
  FROM DUAL
CONNECT BY LEVEL <= 100000 ;

-- 추가 데이터 입력 

BEGIN
      FOR I IN 1..8 LOOP
           INSERT INTO ROWNUM_T2 SELECT * FROM ROWNUM_T2;
           COMMIT;
      END LOOP;
END;
/

-- 각 칼럼에 인덱스 생성 및 통계정보 수집

CREATE INDEX ROWNUM_T2_IDX_01 ON ROWNUM_T2 ( C2 ) ;

CREATE INDEX ROWNUM_T2_IDX_03 ON ROWNUM_T2 (C2,C3) ;
```
